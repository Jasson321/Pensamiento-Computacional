#include <iostream>
using namespace std;
int main ()
{
    int A[4]={4,6,2,3};
    int aux=0;
for (int i=0; i<4; i++)
{
    for (int j=0; j<4; j++)
    {
        if (A[j+1]>A[j]) // invertir el signo para orden ascendente
        {
        aux=A[j];
        A[j]=A[j+1];
        A[j+1]=aux;
        }
    }
}
    for (int i=0; i<4; i++)
    {
        cout<<A[i]<<endl;
    }
return 0;

}
